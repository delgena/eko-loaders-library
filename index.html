Indeterminate Spinner
24px · 3px stroke · 1000ms · LinearEasing
@Composable fun IndeterminateLoader(modifier: Modifier = Modifier) { val transition = rememberInfiniteTransition(label = "loader") val rotation by transition.animateFloat( initialValue = 0f, targetValue = 360f, animationSpec = infiniteRepeatable( animation = tween(1000, easing = LinearEasing) ), label = "rotation" ) Canvas(modifier = modifier.size(24.dp)) { val sw = 3.dp.toPx() val r = (size.minDimension - sw) / 2f rotate(rotation, center) { drawArc( brush = Brush.sweepGradient( 0f to Color(0xFFFFFFFF), 1f to Color(0xFFA5A5A5), center = center ), startAngle = -1f, sweepAngle = 359f, useCenter = false, topLeft = Offset(sw / 2f, sw / 2f), size = Size(r * 2f, r * 2f), style = Stroke(sw, cap = StrokeCap.Round) ) } } }
